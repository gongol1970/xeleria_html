  {key:'inventory_import',source:'Carga general de inventario',slot:'Importación',view:'productos'}

function notificationIsUnread(n){return !n.read_at}
function notificationType(n){return String((n&&n.type)||'').toLowerCase()}
function notificationTarget(n){return String((n&&n.target_view)||'').toLowerCase()}
function notificationGroupKey(n){
  let t=notificationType(n), v=notificationTarget(n);
  if(t.includes('inventory_import')||v.includes('inventory-import'))return 'inventory_import';
  if((t.includes('question')||v.includes('ml-preguntas'))&&t.includes('ml'))return 'ml_questions';
  if(v.includes('ml-preguntas')||t.includes('ml_question'))return 'ml_questions';
  if((t.includes('sale')||t.includes('order')||v.includes('ml-ventas'))&&t.includes('ml'))return 'ml_sales';
  if((t.includes('sale')||t.includes('order')||v.includes('tn-ventas'))&&t.includes('tn'))return 'tn_sales';
  return 'system_other';
}
function notificationGroups(){
  let groups={};
  (state.notifications.items||[]).forEach(n=>{
    let k=notificationGroupKey(n);
    groups[k]=groups[k]||[];
    groups[k].push(n);
  });
  Object.keys(groups).forEach(k=>groups[k].sort((a,b)=>String(b.created_at||'').localeCompare(String(a.created_at||''))));
  return groups;
}
function notificationUnreadCountFor(key){
  let g=notificationGroups()[key]||[];
  return g.filter(notificationIsUnread).length;
}
function notificationImportState(items){
  let n=(items||[])[0];
  if(!n)return 'Sin novedades';
  let t=notificationType(n), sev=String(n.severity||'').toLowerCase();
  if(sev==='warning'||sev==='error'||sev==='bad')return (n.title||'Revisión').replace('Grilla de inventario guardada','Guardado con revisión');
  if(t.includes('finished'))return 'Finalizado';
  if(t.includes('saved'))return 'Guardado';
  if(t.includes('running')||t.includes('started'))return 'Ejecutando';
  return n.title||'Novedad';
}
function renderNotificationsBadge(){
  let total=Number(state.notifications.unread_count||0);
  let nb=document.querySelector('.navBtn[data-view="notificaciones"]');
  if(nb){
    nb.classList.toggle('hasAlert',total>0);
    let s=nb.querySelector('.navAlertBadge');
    if(s)s.textContent=String(total);
  }
}
function notificationIsSale(n){
  let k=notificationGroupKey(n);
  return k==='ml_sales'||k==='tn_sales';
}
function notificationCommercialLine(n){
  n=n||{};
  let p=n.payload||{};
  if(p.commercial_line)return String(p.commercial_line);
  if(p.sale_at_ar&&p.sku_label&&p.product_label){
    return [p.sale_at_ar,p.sku_label,p.product_label,p.quantity_label||'',p.type_label||'',p.shipping_label||''].filter(Boolean).join(' / ');
  }
  let title=String(n.title||'').trim();
  let body=String(n.body||'').trim();
  if(title.includes(' / ')&&(title.includes(' hs')||title.includes('hs')))return title;
  if(body.includes(' / ')&&(body.includes(' hs')||body.includes('hs')))return body;
  return title||body||n.type||'Notificación';
}
function notificationDisplayTitle(n){
  if(notificationIsSale(n))return notificationCommercialLine(n);
  return n.title||n.type||'Notificación';
}
function notificationDisplayMeta(n){
  if(notificationIsSale(n))return '';
  let parts=[];
  if(n.created_at)parts.push(dateAR(n.created_at));
  if(n.body)parts.push(esc(n.body));
  return parts.join(' · ');
}
