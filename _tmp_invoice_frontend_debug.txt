function invoiceButton(o,ch){
  // ARCA_FINAL_CLEANUP_V15
  // Vista compacta: en Ventas alcanza con icono. El detalle queda para ARCA reportes.
  let invId=o.invoice_id||o.arca_invoice_id||'';
  let st=String(o.invoice_status||o.arca_status||'').toLowerCase();
  let hasCae=!!(o.invoice_cae||o.cae||o.arca_cae);
  if(hasCae||st==='issued'){
    let icon='<span class="invoiceStatusIcon issued" title="Facturada">📄</span>';
    return `<div class="invoiceMiniWrap">${invId?`<a target="_blank" href="${url('/admin/arca/facturas/'+encodeURIComponent(invId)+'/pdf')}">${icon}</a>`:icon}</div>`;
  }
  if(st==='draft'||st==='pending_to_invoice'||st==='pending'){
    return `<div class="invoiceMiniWrap"><span class="invoiceStatusIcon draft" title="Pendiente para facturar">📄⚠️</span></div>`;
  }
  return `<div class="invoiceMiniWrap"><span class="invoiceStatusIcon none" title="Sin factura">📄⚠️</span></div>`;
}
function isTnMotoShipping(o){
  let opt=String(o.shipping_option||o.tn_shipping_option||'').toLowerCase();
  return opt.includes('moto')||opt.includes('motomensaj')||opt.includes('mensajería')||opt.includes('mensajeria');
}
function canMarkTnDelivered(o,ch){
  if(ch!=='TN')return false;
  if(!isTnMotoShipping(o))return false;
  let st=String(shippingStateLabel(o,ch)||'').toLowerCase();
  let raw=shippingStatusText(o);
  let already=!!(o.tn_manual_delivered||o.manual_delivered)||st.includes('entregado')||raw.includes('delivered')||raw.includes('manual_delivered');
  if(already)return false;
  return st.includes('en camino')||raw.includes('shipped');
}
function tnDeliveredButton(o,ch){
  if(!canMarkTnDelivered(o,ch))return '';
  let oid=String(o.external_order_id||'');
  if(!oid)return '';
  let label=String(orderNumber(o,ch)||oid).replace(/'/g,'');
  return `<br><button class="secondary smallBtn" onclick="markTnMotoDelivered('${esc(oid)}','${esc(label)}')">Entregado</button>`;
}
